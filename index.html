Data Collection

Create MySQL Database and Tables


CREATE DATABASE BusinessData;
USE BusinessData;

CREATE TABLE Revenue (
    date DATE,
    amount DECIMAL(10, 2)
);

CREATE TABLE Costs (
    date DATE,
    amount DECIMAL(10, 2),
    category VARCHAR(50)
);

CREATE TABLE CustomerFeedback (
    date DATE,
    feedback TEXT,
    rating INT
);

Insert Sample Data

Insert sample data into the tables as shown above.

Data Processing and Analysis

Connect to MySQL Using Python


import mysql.connector
import pandas as pd

conn = mysql.connector.connect(
    host="localhost",
    user="yourusername",
    password="yourpassword",
    database="BusinessData"
)

revenue_df = pd.read_sql("SELECT * FROM Revenue", conn)
costs_df = pd.read_sql("SELECT * FROM Costs", conn)
feedback_df = pd.read_sql("SELECT * FROM CustomerFeedback", conn)

Data Cleaning


revenue_df.dropna(inplace=True)
costs_df.dropna(inplace=True)
feedback_df.dropna(inplace=True)

Data Analysis


total_revenue = revenue_df['amount'].sum()
total_costs = costs_df['amount'].sum()
average_rating = feedback_df['rating'].mean()
roi = (total_revenue - total_costs) / total_costs

print(f"Total Revenue: {total_revenue}")
print(f"Total Costs: {total_costs}")
print(f"Average Customer Rating: {average_rating}")
print(f"ROI: {roi}")
